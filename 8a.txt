#include <stdio.h>
#include <mpi.h>

int main(int argc, char** argv) {
    int rank, size;
    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    int data;
    if (rank == 0)
        data = 100;

    MPI_Bcast(&data, 1, MPI_INT, 0, MPI_COMM_WORLD);
    printf("Process %d received %d via Bcast\n", rank, data);

    if (size >= 2) {
        if (rank == 0) {
            int msg = 200;
            MPI_Send(&msg, 1, MPI_INT, 1, 0, MPI_COMM_WORLD);
        }
        else if (rank == 1) {
            int msg;
            MPI_Recv(&msg, 1, MPI_INT, 0, 0, MPI_COMM_WORLD,
                     MPI_STATUS_IGNORE);
            printf("Process 1 received %d via Send/Recv\n", msg);
        }
    }

    MPI_Finalize();
    return 0;
}
