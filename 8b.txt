#include<stdio.h>
#include<mpi.h>
int main(int argc, char** argv) {

  int rank,size;
  MPI_Init(&argc, &argv);

  MPI_Comm_rank(MPI_COMM_WORLD,&rank);
  MPI_Comm_size(MPI_COMM_WORLD,&size);

  int N=16;
  int elements_per_proc = N/size;

  int a[16];
  int local[elements_per_proc];

  if(rank==0) for(int i=0;i<N;i++) a[i]=i+1;

  MPI_Scatter(a,elements_per_proc,MPI_INT,local,elements_per_proc,MPI_INT,0,MPI_COMM_WORLD);

  int local_sum=0;
  for(int i=0;i<elements_per_proc;i++) local_sum+=local[i];

  int total_sum=0;
  MPI_Reduce(&local_sum,&total_sum,1,MPI_INT,MPI_SUM,0,MPI_COMM_WORLD);

  if(rank==0) printf("Total sum=%d",total_sum);

  MPI_Finalize();
  return 0;
}
