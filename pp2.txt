#include <stdio.h>
#include <omp.h>

int main() {
    int num_steps = 100000;
    double step = 1.0 / (double)num_steps;
    double pi = 0.0;
    
    #pragma omp parallel
    {
        double x, sum = 0.0;
        int i;
        #pragma omp for
        for (i = 0; i < num_steps; i++) {
            x = (i + 0.5) * step;
            sum += 4.0 / (1.0 + x*x);
        }
        #pragma omp critical
        {
            pi += sum * step;
        }
    }
    printf("Computed Pi: %f\n", pi);
    return 0;
}
// for compilation : gcc 2.c -fopenmp -o 2
// for execution : ./2
