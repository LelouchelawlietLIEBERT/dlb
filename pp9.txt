#include <mpi.h>
#include <stdio.h>

#define N 100

int main(int argc,char**argv){
    MPI_Init(&argc,&argv);
    int rank,size;
    MPI_Comm_rank(MPI_COMM_WORLD,&rank);
    MPI_Comm_size(MPI_COMM_WORLD,&size);

    int arr[N],sub[N/size];
    if(rank==0)
        for(int i=0;i<N;i++) arr[i]=1;

    MPI_Scatter(arr,N/size,MPI_INT,sub,N/size,MPI_INT,0,MPI_COMM_WORLD);

    int sum=0;
    for(int i=0;i<N/size;i++) sum+=sub[i];

    int total;
    MPI_Reduce(&sum,&total,1,MPI_INT,MPI_SUM,0,MPI_COMM_WORLD);

    if(rank==0) printf("Total=%d\n",total);
    MPI_Finalize();
}
// execution : mpicc 9.c -o 9
// Run: mpirun -np 2 ./9
